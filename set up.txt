# Create solution
dotnet new sln -n BessPressPortal

# Create WebAPI (backend)
dotnet new webapi -n BessPressPortal.Api
dotnet sln add BessPressPortal.Api

# Create Blazor WASM (frontend) 
dotnet new blazorwasm --auth Individual -n BessPressPortal.Client
dotnet sln add BessPressPortal.Client

# Create shared models project
dotnet new classlib -n BessPressPortal.Shared
dotnet sln add BessPressPortal.Shared


cd BessPressPortal.Api
dotnet add reference ../BessPressPortal.Shared

dotnet add package Microsoft.AspNetCore.Identity.EntityFrameworkCore --version 8.0.0
dotnet add package Microsoft.EntityFrameworkCore.Design --version 8.0.0
dotnet add package Microsoft.EntityFrameworkCore.SqlServer --version 8.0.0
dotnet add package Microsoft.AspNetCore.Authentication.JwtBearer --version 8.0.0
dotnet add package System.IdentityModel.Tokens.Jwt --version 7.0.0
dotnet add package Swashbuckle.AspNetCore
dotnet add package System.Linq.Async



//ASP.NET Core Identity
dotnet ef database drop
// then
rm -r Migrations
dotnet clean
dotnet build
dotnet ef migrations add CreateIdentityTables
dotnet ef database update


cd ../BessPressPortal.Client
dotnet add reference ../BessPressPortal.Shared


dotnet add package Microsoft.AspNetCore.Components.Authorization
dotnet add package Microsoft.AspNetCore.Components.WebAssembly.Authentication
